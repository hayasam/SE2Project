\subsection{Indention}
\begin{enumerate}
	\setcounter{enumi}{7}
	\item Three or four spaces are used for indentation and done so consistently:
	\begin{itemize}
		\item Method 1: \cmark
		\item Method 2: \xmark\\
		Line 169: used TAB instead of spaces
		\code{169}{169}
	\end{itemize}
	\item No tabs are used to indent:
	\begin{itemize}
		\item Method 1: \cmark
		\item Method 2: \xmark\\
		Line 169: used TAB instead of spaces
		\code{169}{169}
	\end{itemize}
\end{enumerate}
